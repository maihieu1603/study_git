git restore --staged filename

#đưa code về commit trước đó

git reset <id commit> --mixed //giữ các thay đổi ở working
git reset <id commit> --soft //giữ các thay đổi ở staged
git reset <id commit> --hard //xóa hết trạng thái giống hệt commit truyền vào
